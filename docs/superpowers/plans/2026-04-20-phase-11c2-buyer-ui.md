### Task 5: Add `scripts/sync-ui-exports.ts` helper

**Files:**
- Create: `scripts/sync-ui-exports.ts`
- Modify: `package.json` — add `ui:sync-exports` script

- [ ] Script regenerates the `exports` map by reading `packages/ui/src/components/*.tsx`. Run it after every `shadcn add`.

```typescript
#!/usr/bin/env bun
import { readdir, readFile, writeFile } from 'node:fs/promises';
import { join } from 'node:path';

const ROOT = new URL('..', import.meta.url).pathname;
const UI_DIR = join(ROOT, 'packages', 'ui');
const COMPONENTS_DIR = join(UI_DIR, 'src', 'components');
const PKG_PATH = join(UI_DIR, 'package.json');

const fixedExports: Record<string, string> = {
  './globals.css': './src/globals.css',
  './tailwind.config': './tailwind.config.ts',
  './cn': './src/utils/cn.ts',
  './hooks/use-toast': './src/hooks/use-toast.ts',
};

const files = (await readdir(COMPONENTS_DIR)).filter(f => f.endsWith('.tsx'));
const dynamic: Record<string, string> = {};
for (const file of files) {
  const name = file.replace(/\.tsx$/, '');
  dynamic[`./${name}`] = `./src/components/${file}`;
}

const pkg = JSON.parse(await readFile(PKG_PATH, 'utf8'));
pkg.exports = { ...fixedExports, ...dynamic };
await writeFile(PKG_PATH, JSON.stringify(pkg, null, 2) + '\n');
console.log(`synced ${Object.keys(dynamic).length} component exports`);
```

### Task 11: `parse-list-query.ts`

**Files:**
- Create: `apps/app/lib/parse-list-query.ts`

```typescript
export interface ListQuery {
  page: number;
  per: number;
  skip: number;
  take: number;
  sort: string;
  filters: Record<string, string>;
}

export function parseListQuery(
  searchParams: Record<string, string | string[] | undefined>,
  opts: { defaultPer?: number; maxPer?: number; knownFilters?: string[] } = {}
): ListQuery {
  const defaultPer = opts.defaultPer ?? 25;
  const maxPer = opts.maxPer ?? 100;
  const page = Math.max(1, Number(searchParams.page ?? 1));
  const per = Math.min(maxPer, Math.max(1, Number(searchParams.per ?? defaultPer)));
  const skip = (page - 1) * per;
  const sort = String(searchParams.sort ?? '-createdAt');
  const filters: Record<string, string> = {};
  for (const k of opts.knownFilters ?? []) {
    const v = searchParams[k];
    if (typeof v === 'string' && v.length > 0) filters[k] = v;
  }
  return { page, per, skip, take: per, sort, filters };
}
```

### Task 13: `PagePagination`

**Files:**
- Create: `apps/app/components/shared/page-pagination.tsx`

```tsx
import Link from 'next/link';
import { cn } from '@sendero/ui/cn';

export function PagePagination({
  page,
  totalPages,
  baseUrl,
  searchParams,
}: {
  page: number;
  totalPages: number;
  baseUrl: string;
  searchParams: Record<string, string | undefined>;
}) {
  function urlFor(p: number) {
    const params = new URLSearchParams();
    for (const [k, v] of Object.entries(searchParams)) {
      if (v !== undefined && k !== 'page') params.set(k, v);
    }
    params.set('page', String(p));
    return `${baseUrl}?${params.toString()}`;
  }

  if (totalPages <= 1) return null;

  return (
    <div className="flex items-center justify-between border-t pt-4 text-sm">
      <span className="text-neutral-500">Page {page} of {totalPages}</span>
      <div className="flex gap-2">
        {page > 1 && (
          <Link href={urlFor(page - 1)} className="px-3 py-1 rounded hover:bg-neutral-100">
            ← Prev
          </Link>
        )}
        {page < totalPages && (
          <Link href={urlFor(page + 1)} className="px-3 py-1 rounded hover:bg-neutral-100">
            Next →
          </Link>
        )}
      </div>
    </div>
  );
}
```

- [ ] **Step 3: `trips/page.tsx`**

```tsx
import { auth } from '@sendero/auth/server';
import { prisma } from '@sendero/database';
import { notFound } from 'next/navigation';
import { TripsTable } from '@/components/trips/trips-table';
import { PrefundSheet } from '@/components/trips/prefund-sheet/prefund-sheet';
import { PagePagination } from '@/components/shared/page-pagination';
import { EmptyState } from '@/components/shared/empty-state';
import { parseListQuery } from '@/lib/parse-list-query';
import { Button } from '@sendero/ui/button';
import Link from 'next/link';

export default async function TripsPage({
  searchParams,
}: {
  searchParams: Promise<Record<string, string | undefined>>;
}) {
  const { orgId } = await auth();
  const params = await searchParams;
  const tenant = await prisma.tenant.findUnique({ where: { clerkOrgId: orgId! } });
  if (!tenant) notFound();

  const q = parseListQuery(params, { knownFilters: ['status'] });
  const where = { tenantId: tenant.id, ...(q.filters.status ? { status: q.filters.status } : {}) };
  const [trips, total] = await Promise.all([
    prisma.trip.findMany({
      where,
      skip: q.skip,
      take: q.take,
      orderBy: { createdAt: 'desc' },
      select: { id: true, summary: true, budgetUsd: true, status: true, createdAt: true },
    }),
    prisma.trip.count({ where }),
  ]);
  const totalPages = Math.max(1, Math.ceil(total / q.per));
  const showSheet = params.sheet === 'new';

  return (
    <>
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-semibold">Trips</h1>
        <Button asChild>
          <Link href="/app/trips?sheet=new">New trip</Link>
        </Button>
      </div>

      {trips.length === 0 ? (
        <EmptyState
          title="No trips yet"
          description="Create a prefunded trip to send to a traveler."
          cta={{ label: 'New trip', href: '/app/trips?sheet=new' }}
        />
      ) : (
        <>
          <TripsTable trips={trips.map(t => ({ ...t, budgetUsd: t.budgetUsd.toString() }))} />
          <PagePagination page={q.page} totalPages={totalPages} baseUrl="/app/trips" searchParams={params} />
        </>
      )}

      <PrefundSheet open={showSheet} />
    </>
  );
}
```

```tsx
'use client';

import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { Button } from '@sendero/ui/button';
import { Input } from '@sendero/ui/input';
import { Label } from '@sendero/ui/label';
import { Checkbox } from '@sendero/ui/checkbox';
import { useState } from 'react';
import { toast } from '@sendero/ui/hooks/use-toast';

const schema = z.object({
  budgetUsdc: z.string().regex(/^\d+(\.\d{1,6})?$/, 'Decimal USDC (e.g. 500.00)'),
  guestEmail: z.string().email(),
  guestName: z.string().max(80).optional(),
  tripSummary: z.string().max(200).optional(),
  expiresInDays: z.number().int().min(1).max(365).default(30),
  require2fa: z.boolean().default(true),
});

type FormValues = z.infer<typeof schema>;

export function PrefundForm({
  onSuccess,
}: {
  onSuccess: (r: { tripId: string; guestLink: string; claimCode?: string; onchainCalls: unknown }) => void;
}) {
  const form = useForm<FormValues>({
    resolver: zodResolver(schema),
    defaultValues: { expiresInDays: 30, require2fa: true },
  });
  const [submitting, setSubmitting] = useState(false);

  async function onSubmit(values: FormValues) {
    setSubmitting(true);
    try {
      const res = await fetch('/api/guest/invite', {
        method: 'POST',
        headers: { 'content-type': 'application/json' },
        body: JSON.stringify(values),
      });
      if (!res.ok) {
        const err = await res.json().catch(() => ({}));
        toast({ title: 'Invite failed', description: String(err.message ?? res.statusText), variant: 'destructive' });
        return;
      }
      const data = await res.json();
      onSuccess({
        tripId: data.tripId,
        guestLink: data.guestLink,
        claimCode: data.claimCode,
        onchainCalls: data.onchainCalls,
      });
    } finally {
      setSubmitting(false);
    }
  }

  return (
    <form onSubmit={form.handleSubmit(onSubmit)} className="flex flex-col gap-4 py-4">
      <div>
        <Label htmlFor="budgetUsdc">Budget (USDC)</Label>
        <Input id="budgetUsdc" type="text" inputMode="decimal" placeholder="500.00" {...form.register('budgetUsdc')} />
        {form.formState.errors.budgetUsdc && <p className="text-xs text-red-600">{form.formState.errors.budgetUsdc.message}</p>}
      </div>
      <div>
        <Label htmlFor="guestEmail">Guest email</Label>
        <Input id="guestEmail" type="email" {...form.register('guestEmail')} />
        {form.formState.errors.guestEmail && <p className="text-xs text-red-600">{form.formState.errors.guestEmail.message}</p>}
      </div>
      <div>
        <Label htmlFor="guestName">Guest name (optional)</Label>
        <Input id="guestName" {...form.register('guestName')} />
      </div>
      <div>
        <Label htmlFor="tripSummary">Trip summary (optional)</Label>
        <Input id="tripSummary" placeholder="SFO → LHR, May 3-10" {...form.register('tripSummary')} />
      </div>
      <div className="flex items-center gap-2">
        <Checkbox id="require2fa" checked={form.watch('require2fa')} onCheckedChange={v => form.setValue('require2fa', Boolean(v))} />
        <Label htmlFor="require2fa" className="cursor-pointer">Require 6-digit code (2FA)</Label>
      </div>
      <Button type="submit" disabled={submitting}>{submitting ? 'Sending…' : 'Send invite'}</Button>
    </form>
  );
}

'use client';

import { Button } from '@sendero/ui/button';
import { Download } from 'lucide-react';

export function DownloadPdfButton({ invoiceId }: { invoiceId: string }) {
  async function download() {
    const res = await fetch(`/api/invoices/${invoiceId}/pdf`);
    if (!res.ok) return;
    const blob = await res.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `invoice-${invoiceId}.pdf`;
    document.body.appendChild(a);
    a.click();
    a.remove();
    URL.revokeObjectURL(url);
  }
  return (
    <Button onClick={download} variant="outline">
      <Download className="mr-2 h-4 w-4" /> Download PDF
    </Button>
  );
}

import { InvoiceDetailView } from '@/components/invoices/invoice-detail-view';

export default async function InvoiceDetailPage({

}) {

  return (
    <div className="space-y-6">
      <InvoiceDetailView invoice={invoice} />
      <DownloadPdfButton invoiceId={invoice.id} />
    </div>

// actions.ts
'use server';

import { auth, clerkClient } from '@sendero/auth/server';
import { prisma } from '@sendero/database';
import { z } from 'zod';
import { revalidatePath } from 'next/cache';

const BillingSchema = z.object({
  legalName: z.string().max(160).optional(),
  billingContactEmail: z.string().email().optional(),
  billingAddress: z.object({
    line1: z.string().optional(),
    line2: z.string().optional(),
    city: z.string().optional(),
    region: z.string().optional(),
    postalCode: z.string().optional(),
    country: z.string().optional(),
  }).optional(),
  taxId: z.string().max(40).optional(),
});

export async function updateTenantBillingAction(raw: unknown) {
  const { has, orgId } = await auth();
  if (!has({ role: 'org:admin' })) return { error: 'forbidden' };
  if (!orgId) return { error: 'no_org' };

  const parsed = BillingSchema.safeParse(raw);
  if (!parsed.success) return { error: 'invalid_input', issues: parsed.error.issues };

  const tenant = await prisma.tenant.update({
    where: { clerkOrgId: orgId },
    data: parsed.data,
  });

  const client = await clerkClient();
  await client.organizations.updateOrganization(orgId, {
    publicMetadata: {
      legalName: parsed.data.legalName ?? tenant.legalName,
      taxId: parsed.data.taxId ?? tenant.taxId,
    },
  });

  revalidatePath('/app/settings/billing');
  return { ok: true };
}
```
